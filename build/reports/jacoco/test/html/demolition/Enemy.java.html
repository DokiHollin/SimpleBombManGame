package demolition;

import processing.core.PApplet;
import processing.core.PImage;

public class Enemy implements PlayerLife{
    protected int x;
    protected int y;
    protected PImage sprite;
    protected boolean isAlive;
    protected int moveSpeed;

    protected int timeCounter = 0;
    protected int EnemyLife;

    public void setPEnemyLife(int EnemyLife) {
        this.EnemyLife = EnemyLife;
    }

    public Enemy(int x, int y, PImage sprite){
        this.x = x;
        this.y = y;
        this.sprite = sprite;
        this.moveSpeed = 32;
        this.isAlive = true;
        this.EnemyLife = 1;
    }

    public void tick(){
        
    }

    public void draw(PApplet app){
        app.image(this.sprite, this.x, this.y-getHeight()/2+8);
    }

    public int getHeight(){
        return this.sprite.height;
    }

    public int getWidth(){
        return this.sprite.width;
    }

    public void reduceEnemyLife(){
        this.EnemyLife --;
    }

    public int getEnemyLife() {
        return EnemyLife;
    }

    public boolean isAlive(){
        return this.isAlive;
    }

    public void setSprite(PImage sprite) {
        this.sprite = sprite;
    }

    public int getX(){
        return this.x;
    }

    public int getY(){
        return this.y;
    }
    
    public void dead(){
        this.isAlive = false;
    }
    
}
